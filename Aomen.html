<script>

    // 对应图2生肖规则

    const zodiacRules = {

        "鼠": [7, 19, 31, 43], "牛": [6, 18, 30, 42], "虎": [5, 17, 29, 41],

        "兔": [4, 16, 28, 40], "龙": [3, 15, 27, 39], "蛇": [2, 14, 26, 38],

        "马": [1, 13, 25, 37, 49], "羊": [12, 24, 36, 48], "猴": [11, 23, 35, 47],

        "鸡": [10, 22, 34, 46], "狗": [9, 21, 33, 45], "猪": [8, 20, 32, 44]

    };


    const numToZodiac = {};

    for (let zodiac in zodiacRules) {

        zodiacRules[zodiac].forEach(num => {

            numToZodiac[num] = zodiac;

        });

    }


    let amounts = {}; 

    let currentFilter = 'all';


    function init() {

        const zodiacBar = document.getElementById('zodiacBar');

        const grid = document.getElementById('numberGrid');


        Object.keys(zodiacRules).forEach(z => {

            const btn = document.createElement('div');

            btn.className = 'zodiac-btn'; 

            btn.innerText = z;

            btn.onclick = () => quickInputZodiac(z);

            zodiacBar.appendChild(btn);

        });


        for (let i = 1; i <= 49; i++) {

            amounts[i] = 0;

            const card = document.createElement('div');

            card.className = 'number-card';

            card.id = `card-${i}`;

            

            const color = [1,2,7,8,12,13,18,19,23,24,29,30,34,35,40,45,46].includes(i) ? 'red' : ([3,4,9,10,14,15,20,25,26,31,36,41,42,47,48].includes(i) ? 'blue' : 'green');

            const padNum = i < 10 ? '0' + i : i;

            const zodiacName = numToZodiac[i];


            card.innerHTML = `

                <div class="ball-box">

                    <div class="ball ${color}">${padNum}</div>

                    <div class="zodiac-tag">(${zodiacName})</div>

                </div>

                <input type="number" class="amount-input" placeholder="金额" id="input-${i}" oninput="updateAmount(${i}, this.value)" onclick="event.stopPropagation()">

            `;

            card.onclick = () => document.getElementById(`input-${i}`).focus();

            grid.appendChild(card);

        }

    }


    function updateAmount(num, val) {

        const amt = parseFloat(val) || 0;

        amounts[num] = amt;

        

        const card = document.getElementById(`card-${num}`);

        if (amt > 0) {

            card.classList.add('selected');

        } else {

            card.classList.remove('selected');

        }

        calculate();

    }


    function quickInputZodiac(zodiac) {

        const val = prompt(`请输入生肖【${zodiac}】所有对应号码的快捷下注金额：`, "");

        if (val === null || val.trim() === "") return;

        

        const amt = parseFloat(val) || 0;

        const nums = zodiacRules[zodiac];


        nums.forEach(n => {

            amounts[n] = amt;

            const input = document.getElementById(`input-${n}`);

            const card = document.getElementById(`card-${n}`);

            

            if (amt > 0) {

                input.value = amt;

                card.classList.add('selected');

            } else {

                input.value = '';

                card.classList.remove('selected');

            }

        });

        calculate();

    }


    function setFilter(f) {

        currentFilter = f;

        document.querySelectorAll('.filter-btn').forEach(b => {

            const isMatch = (f === 'all' && b.innerText === '全部') || 

                            (f === 'odd' && b.innerText === '单数') || 

                            (f === 'even' && b.innerText === '双数');

            b.classList.toggle('active', isMatch);

        });

        calculate();

    }


    function calculate() {

        let total = 0;

        let html = '';

        

        for (let i = 1; i <= 49; i++) {

            const isSelected = document.getElementById(`card-${i}`).classList.contains('selected');

            if (!isSelected || amounts[i] <= 0) continue;


            const isOdd = i % 2 !== 0;

            if (currentFilter === 'odd' && !isOdd) continue;

            if (currentFilter === 'even' && isOdd) continue;


            total += amounts[i];

            const padNum = i < 10 ? '0' + i : i;

            html += `

                <div class="summary-item">

                    <span>号码 <strong>${padNum}</strong> (${numToZodiac[i]}) - ${isOdd ? '单数' : '双数'}</span>

                    <span><strong>${amounts[i]}</strong> 元</span>

                </div>

            `;

        }

        document.getElementById('totalSum').innerText = total.toFixed(2);

        document.getElementById('summaryList').innerHTML = html || '<div style="color:#999; text-align:center; padding:10px;">没有符合筛选条件的投注</div>';

    }


    function resetAll() {

        amounts = {};

        document.querySelectorAll('.number-card').forEach(c => {

            c.classList.remove('selected');

            c.querySelector('input').value = '';

        });

        calculate();

    }


    init();

</script>
